library(dplyr); library(ggplot2)

emis <- readRDS("summarySCC_PM25.rds")
emis <- transform(emis, year=factor(year), type=factor(type))

totalPM2.5 <- tapply(emis$Emissions, emis$year, sum)
totDat <- data.frame(totalPM2.5)
totDat$year <- row.names(totDat)
row.names(totDat) <- NULL

par(mar=c(4,4,2,1))
barplot(totDat$totalPM2.5, names.arg=totDat$year, col.axis="darkblue",  
                        col="wheat", border = "darkblue", las=0)
title(main="total pm2.5 - all sources", xlab="year", ylab="total pm2.5", font.lab=2, col.lab="red", col.main="red")

totalPM2.5balt <- tapply(emis$Emissions[emis$fips=="24510"], 
                         emis$year[emis$fips=="24510"], sum)
totDatBalt <- data.frame(totalPM2.5balt)
totDatBalt$year <- row.names(totDatBalt)
row.names(totDatBalt) <- NULL

par(mar=c(4,4,2,1))
barplot(totDatBalt$totalPM2.5balt, names.arg=totDatBalt$year, las=0)
title(main="total pm2.5 - Baltimore City - all sources", xlab="year", ylab="total pm2.5", 
        font.lab=2, col.lab="blue", col.main="blue")

baltimore <- emis %>% filter(fips=="24510") %>% droplevels

baltimore <- aggregate(Emissions ~ year + type, baltimore, sum)


g<- ggplot(baltimore, aes(year,Emissions))

g <- g + geom_bar(width=.5, stat="identity", aes(fill=year)) + 
        facet_grid(.~type) + aes(color=year) + 
        labs(title="Total pm2.5 for the four years per source") + 
        theme(plot.title=element_text(color="blue", size=14, hjust=0.5), 
                        axis.title.x=element_text(color="blue", size=12), 
                        axis.title.y=element_text(color="blue", size=12))
print(g)

sccd <- readRDS("Source_Classification_Code.rds")

source_coal <- grepl("coal", sccd$Short.Name, ignore.case=TRUE) |
                grepl("coal", sccd$EI.Sector, ignore.case=TRUE)
source_coal_sccd <- sccd[source_coal, ]
source_coal_emis <- emis[emis$SCC %in% source_coal_sccd$SCC, ]
source_coal_annualEmis <- aggregate(Emissions ~ year, source_coal_emis, sum)
source_coal_annualEmis$year <- as.factor(source_coal_annualEmis$year)

g <- ggplot(source_coal_annualEmis, aes(year, Emissions))
g <- g + geom_bar(width=0.5, stat="identity", aes(fill=year))
g <- g + ggtitle("Total PM2.5 emissions from coal combustion-related sources")
g <- g + theme(axis.text=element_text(color="black", size=10))
g <- g + theme(axis.title.x=element_text(color="red", size=11), 
               axis.title.y=element_text(color="red", size=11),
               plot.title=element_text(color="red", hjust=0.5))
print(g)

onRoadBalt <- emis[emis$type=="ON-ROAD" & emis$fips=="24510", ]
onRoadBaltAnnual <- aggregate(Emissions ~ year, onRoadBalt, sum)
onRoadBaltAnnual$year <- as.factor(onRoadBaltAnnual$year)

g <- ggplot(onRoadBaltAnnual, aes(year, Emissions))
g <- g + geom_bar(width=0.5, stat="identity", aes(fill=year))
g <- g + ggtitle("Total PM2.5 emissions from vehicle sources - Baltimore")
g <- g + theme(axis.text=element_text(color="black", size=10))
g <- g + theme(axis.title.x=element_text(color="red", size=11), 
               axis.title.y=element_text(color="red", size=11),
               plot.title=element_text(color="red", hjust=0.5))
print(g)

balt_losAngl <- emis[emis$type=="ON-ROAD" & (emis$fips=="24510" | emis$fips=="06037"),]
balt_losAngl_annual <- aggregate(Emissions ~ fips + year, balt_losAngl, sum)
balt_losAngl_annual$fips[balt_losAngl_annual$fips=="24510"]="Baltimore"
balt_losAngl_annual$fips[balt_losAngl_annual$fips=="06037"]="Los Angleles"

balt_losAngl_annual = transform(balt_losAngl_annual, fips=factor(fips), year=factor(year))

g <- ggplot(balt_losAngl_annual, aes(year, Emissions))
g <- g + geom_bar(width=0.5, stat="identity", aes(fill=year))
g <- g + ggtitle("Total PM2.5 emissions from vehicle sources - Baltimore vs. Los Angeles")
g <- g + facet_grid(.~fips)
g <- g + theme(axis.text=element_text(color="black", size=10))
g <- g + theme(axis.title.x=element_text(color="red", size=11), 
               axis.title.y=element_text(color="red", size=11),
               plot.title=element_text(color="red", hjust=0.5))
print(g)
